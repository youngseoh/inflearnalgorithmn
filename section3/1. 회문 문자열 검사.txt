import sys
sys.stdin= open("input.txt","rt")
n= int(input())
for i in range(n):
    s=input()
    s=s.upper() 
    size=len(s)
    for j in range(size//2):
        if s[j]!=s[-1-j]:
            print("#%d NO" %(i+1))
            break
    else:
        print("#%d YES" %(i+1))

for i in range( n):
    s= input()
    s=s.upper()
    if s==s[::-1]:# 맨뒷부터 -1씩 slice 해라
        print("#%d YES" %(i+1))
    else:
        print("#%d No" %(i+1))
